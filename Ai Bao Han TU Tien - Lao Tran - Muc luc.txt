//supported function to extract the chapter list from m array (`arr` param) that is in `rangeStr` to put into sTit of corresponding audio chapter
// for example: extractChapters(m, "10-12") will return "10: Đan Đỉnh phong liền mùi đều là thơm; 11: Bái sư học nghệ; 12: Cử trọng nhược khinh, cử khinh nhược trọng"
function extractChapters(m, range) {
    let start, end;

    if (range.includes("-")) {
        [start, end] = range.split("-").map(Number);
    } else {
        start = end = Number(range);
    }

    k = m.filter(item => {
        const match = item.match(/^Chương\s+(\d+)/i);
        if (!match) return false;

        const chapterNumber = Number(match[1]);
        return chapterNumber >= start && chapterNumber <= end;
    });
	
	return k.map( t => t = t.replace(/^Chương\s*/,"") ).join("; ");
}

//main code
s.split("\n").forEach(r => { //go through each mapping line
	k = r.split(" "); // k[1] is book-range
	h = c[Number(k[0])-1]; // audio chapter index
	h.tit = `Tập ${k[0]} (Chương ${k[1]})`;
	h.sTit = extractChapters(m, k[1]);
}); copy(c)
